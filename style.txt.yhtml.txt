<script>
/* ---------- particle background on canvas ---------- */
(function(){
  const canvas = document.getElementById('bg-canvas');
  if(!canvas) return;
  const ctx = canvas.getContext('2d');
  let w = canvas.width = innerWidth;
  let h = canvas.height = innerHeight;
  const DPR = Math.max(1, window.devicePixelRatio || 1);
  canvas.width = innerWidth * DPR;
  canvas.height = innerHeight * DPR;
  canvas.style.width = innerWidth + 'px';
  canvas.style.height = innerHeight + 'px';
  ctx.scale(DPR, DPR);

  const particles = [];
  const COUNT = Math.round((w*h)/(1400*1400) * 120) + 30; // scale with screen
  for(let i=0;i<COUNT;i++){
    particles.push({
      x: Math.random()*w,
      y: Math.random()*h,
      r: 0.6 + Math.random()*2.2,
      vx: (Math.random()-0.5)*0.2,
      vy: (Math.random()-0.5)*0.2,
      alpha: 0.05 + Math.random()*0.25
    });
  }

  function resize(){
    w = canvas.width = innerWidth * DPR;
    h = canvas.height = innerHeight * DPR;
    canvas.style.width = innerWidth + 'px';
    canvas.style.height = innerHeight + 'px';
    ctx.setTransform(DPR,0,0,DPR,0,0);
  }
  addEventListener('resize', ()=>{ resize(); });

  let t = 0;
  function loop(){
    t += 0.007;
    ctx.clearRect(0,0,innerWidth,innerHeight);

    // gentle radial vignette
    const grd = ctx.createLinearGradient(0,0,0,innerHeight);
    grd.addColorStop(0, 'rgba(10,18,22,0.16)');
    grd.addColorStop(1, 'rgba(2,6,8,0.5)');
    ctx.fillStyle = grd;
    ctx.fillRect(0,0,innerWidth,innerHeight);

    // draw particles
    for(let p of particles){
      p.x += p.vx + Math.cos(t + p.x*0.0004)*0.08;
      p.y += p.vy + Math.sin(t + p.y*0.0005)*0.08;
      if(p.x < -20) p.x = innerWidth + 20;
      if(p.x > innerWidth + 20) p.x = -20;
      if(p.y < -20) p.y = innerHeight + 20;
      if(p.y > innerHeight + 20) p.y = -20;

      ctx.beginPath();
      ctx.fillStyle = 'rgba(180,240,255,' + (p.alpha*0.9) + ')';
      ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
      ctx.fill();
    }

    requestAnimationFrame(loop);
  }

  // respect reduced motion
  const reduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
  if(!reduced) loop();
})();

/* ---------- basic UI: year, play buttons, modals ---------- */
document.getElementById('year').textContent = new Date().getFullYear();

// Play/pause buttons (toggle + pause other videos)
document.querySelectorAll('.btn-play').forEach(btn=>{
  btn.addEventListener('click', ()=>{
    const id = btn.dataset.video;
    // If inline video with no data-video (submit/call to contact) ignore
    if(!id) return;
    const v = document.getElementById(id);
    if(!v) return;
    // pause others
    document.querySelectorAll('.card video').forEach(o => { if(o !== v) o.pause(); });
    if(v.paused){ v.play(); btn.textContent = '⏸ Pause'; btn.setAttribute('aria-pressed','true'); }
    else { v.pause(); btn.textContent = '▶ Play'; btn.setAttribute('aria-pressed','false'); }
    v.onended = ()=>{ btn.textContent = '▶ Play'; btn.setAttribute('aria-pressed','false'); };
  });
});

// keep play button in sync with native controls
document.querySelectorAll('.card video').forEach(v=>{
  v.addEventListener('play', ()=> {
    const btn = document.querySelector('.btn-play[data-video="'+v.id+'"]');
    if(btn){ btn.textContent = '⏸ Pause'; btn.setAttribute('aria-pressed','true'); }
  });
  v.addEventListener('pause', ()=> {
    const btn = document.querySelector('.btn-play[data-video="'+v.id+'"]');
    if(btn){ btn.textContent = '▶ Play'; btn.setAttribute('aria-pressed','false'); }
  });
  v.addEventListener('ended', ()=> {
    const btn = document.querySelector('.btn-play[data-video="'+v.id+'"]');
    if(btn){ btn.textContent = '▶ Play'; btn.setAttribute('aria-pressed','false'); }
  });
});

// Modal open/close - View More and teaser
document.querySelectorAll('.btn-view, .btn-open').forEach(b=>{
  b.addEventListener('click', ()=> {
    const m = b.dataset.modal;
    if(m) openModal(m);
  });
});

function openModal(id){
  const modal = document.getElementById(id);
  if(!modal) return;
  // pause all card videos
  document.querySelectorAll('.card video').forEach(v => v.pause());
  modal.classList.add('open');
  modal.setAttribute('aria-hidden','false');
}

function closeModal(id){
  const modal = document.getElementById(id);
  if(!modal) return;
  modal.classList.remove('open');
  modal.setAttribute('aria-hidden','true');
  // pause videos inside modal when closed
  modal.querySelectorAll('video').forEach(v => v.pause());
}

// clicking outside closes
document.querySelectorAll('.modal').forEach(m => {
  m.addEventListener('click', (e) => {
    if(e.target === m) closeModal(m.id);
  });
});

// Esc closes any open modal
window.addEventListener('keydown', (e) => {
  if(e.key === 'Escape'){
    document.querySelectorAll('.modal.open').forEach(m => closeModal(m.id));
  }
});
